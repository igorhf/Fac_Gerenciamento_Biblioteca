2º PASSO

create table usuario(
id int primary key auto_increment,
nome varchar(150),
cpf varchar(18)
);
create table tipo(
id int primary key auto_increment,
nome varchar(200)
);
create table livros(
id int primary key auto_increment,
nome varchar(200),
tipo_livro varchar(200)
);
create table associacao(
id int primary key auto_increment,
idUsuario int,
idLivros int,
foreign key(idUsuario)references usuario(id),
foreign key(idLivros)references livros(id)
);
